/* ===================== 1.a Reverse File using lseek ===================== */
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "Usage: %s <filename>\n", argv[0]);
        return 1;
    }
    int fd = open(argv[1], O_RDONLY);
    int file_size = lseek(fd, 0, SEEK_END);
    for (int i = 1; i <= file_size; i++) {
        lseek(fd, -i, SEEK_END);
        char c;
        read(fd, &c, 1);
        putchar(c);
    }
    printf("\n");
    close(fd);
    return 0;
}

/* ===================== 1.b Parent Child File Offset ===================== */
#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/wait.h>
int main() {
    int fd = open("t2.txt", O_RDWR);
    pid_t pid = fork();
    if (pid == 0) {
        char buffer[10];
        read(fd, buffer, 5);
        buffer[5] = '\0';
        printf("Child read: %s\n", buffer);
    } else {
        wait(NULL);
        char buffer[10];
        read(fd, buffer, 5);
        buffer[5] = '\0';
        printf("Parent read: %s\n", buffer);
    }
    close(fd);
    return 0;
}

/* ===================== 2.a stat structure ===================== */
#include <stdio.h>
#include <sys/stat.h>
int main(int argc, char *argv[]) {
    struct stat file_stat;
    stat(argv[1], &file_stat);
    printf("Size: %lld\n", (long long)file_stat.st_size);
    printf("Permissions: %o\n", file_stat.st_mode & 0777);
    printf("Links: %ld\n", (long)file_stat.st_nlink);
    printf("UID: %ld GID: %ld\n", (long)file_stat.st_uid, (long)file_stat.st_gid);
    return 0;
}

/* ===================== 2.b system() simulation ===================== */
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
int my_sys(const char *cm){
    pid_t pid = fork();
    if(pid==0){
        execl("/bin/sh","sh","-c",cm,(char *)NULL);
        exit(1);
    }
    int st;
    waitpid(pid,&st,0);
    return WEXITSTATUS(st);
}
int main(){
    int res = my_sys("ls -l");
    printf("exit status %d\n",res);
}

/* ===================== 3.a Remove empty files ===================== */
#include <stdio.h>
#include <dirent.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>
void remove_empty_files(const char *directory) {
    struct dirent *entry;
    DIR *dp = opendir(directory);
    while ((entry = readdir(dp)) != NULL) {
        if (!strcmp(entry->d_name,".") || !strcmp(entry->d_name,"..")) continue;
        char path[1024];
        sprintf(path,"%s/%s",directory,entry->d_name);
        int fd = open(path,O_RDONLY);
        if(lseek(fd,0,SEEK_END)==0){
            unlink(path);
            printf("Removed: %s\n",path);
        }
        close(fd);
    }
    closedir(dp);
}
int main(int argc,char *argv[]){
    remove_empty_files(argv[1]);
}

/* ===================== 3.b ls -li ===================== */
#include <stdio.h>
#include <dirent.h>
#include <sys/stat.h>
#include <time.h>
int main(int argc,char *argv[]){
    DIR *dp = opendir((argc>1)?argv[1]:".");
    struct dirent *d;
    struct stat m;
    while((d=readdir(dp))){
        stat(d->d_name,&m);
        printf("%ld %o %ld %s", m.st_ino,m.st_mode,(long)m.st_uid,ctime(&m.st_atime));
    }
}
